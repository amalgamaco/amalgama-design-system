Person Card

Tarjeta de persona para directorios de empleados y candidatos.

Instalacion

Copia person-card.tsx a tu proyecto en components/ui/person-card.tsx

import * as React from "react"
import { cn } from "@/lib/utils"

const PeopleGrid = React.forwardRef<
  HTMLDivElement,
  React.HTMLAttributes<HTMLDivElement>
>(({ className, ...props }, ref) => (
  <div ref={ref} className={cn("people-grid", className)} {...props} />
))
PeopleGrid.displayName = "PeopleGrid"

export interface PersonCardProps extends React.HTMLAttributes<HTMLDivElement> {
  initials: string
  name: string
  role: string
}

const PersonCard = React.forwardRef<HTMLDivElement, PersonCardProps>(
  ({ className, initials, name, role, ...props }, ref) => (
    <div ref={ref} className={cn("person-card", className)} {...props}>
      <div className="person-avatar">{initials}</div>
      <div className="person-info">
        <div className="person-name">{name}</div>
        <div className="person-role">{role}</div>
      </div>
    </div>
  )
)
PersonCard.displayName = "PersonCard"

export { PeopleGrid, PersonCard }
