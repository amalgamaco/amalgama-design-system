Kanban

Tablero de columnas para gestionar procesos de seleccion.

Instalacion

Copia kanban.tsx a tu proyecto en components/ui/kanban.tsx

import * as React from "react"
import { cn } from "@/lib/utils"

const KanbanBoard = React.forwardRef<
  HTMLDivElement,
  React.HTMLAttributes<HTMLDivElement>
>(({ className, ...props }, ref) => (
  <div ref={ref} className={cn("kanban-board", className)} {...props} />
))
KanbanBoard.displayName = "KanbanBoard"

export interface KanbanColumnProps
  extends React.HTMLAttributes<HTMLDivElement> {
  title: string
  count?: number
}

const KanbanColumn = React.forwardRef<HTMLDivElement, KanbanColumnProps>(
  ({ className, title, count, children, ...props }, ref) => (
    <div ref={ref} className={cn("kanban-column", className)} {...props}>
      <div className="kanban-column-header">
        {title}
        {count !== undefined && <span className="count">{count}</span>}
      </div>
      <div className="kanban-column-body">{children}</div>
    </div>
  )
)
KanbanColumn.displayName = "KanbanColumn"

export interface KanbanCardProps extends React.HTMLAttributes<HTMLDivElement> {
  title: string
  meta?: string
}

const KanbanCard = React.forwardRef<HTMLDivElement, KanbanCardProps>(
  ({ className, title, meta, children, ...props }, ref) => (
    <div ref={ref} className={cn("kanban-card", className)} {...props}>
      <div className="kanban-card-title">{title}</div>
      {meta && <div className="kanban-card-meta">{meta}</div>}
      {children}
    </div>
  )
)
KanbanCard.displayName = "KanbanCard"

export { KanbanBoard, KanbanColumn, KanbanCard }
