Vacancy Card

Tarjeta de vacante con informacion del puesto y asignados.

Instalacion

Copia vacancy-card.tsx a tu proyecto en components/ui/vacancy-card.tsx

import * as React from "react"
import { cn } from "@/lib/utils"

const VacanciesList = React.forwardRef<
  HTMLDivElement,
  React.HTMLAttributes<HTMLDivElement>
>(({ className, ...props }, ref) => (
  <div ref={ref} className={cn("vacancies-list", className)} {...props} />
))
VacanciesList.displayName = "VacanciesList"

export interface VacancyCardProps extends React.HTMLAttributes<HTMLDivElement> {
  icon?: React.ReactNode
  name: string
  badge?: React.ReactNode
  meta?: string[]
  age?: string
}

const VacancyCard = React.forwardRef<HTMLDivElement, VacancyCardProps>(
  ({ className, icon, name, badge, meta, age, children, ...props }, ref) => (
    <div ref={ref} className={cn("vacancy-card", className)} {...props}>
      {icon && <div className="vacancy-icon">{icon}</div>}
      <div className="vacancy-info">
        <div className="vacancy-name">
          {name}
          {badge}
        </div>
        {meta && meta.length > 0 && (
          <div className="vacancy-meta">
            {meta.map((item, i) => (
              <React.Fragment key={i}>
                {i > 0 && <span className="meta-dot" />}
                {item}
              </React.Fragment>
            ))}
          </div>
        )}
        {children}
      </div>
      {age && (
        <div className="vacancy-right">
          <div className="vacancy-age">{age}</div>
        </div>
      )}
    </div>
  )
)
VacancyCard.displayName = "VacancyCard"

export { VacanciesList, VacancyCard }
